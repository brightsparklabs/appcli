 195    """
 196    Uses Docker Compose to orchestrate containers.
 197    """
 198
 199    def __init__(
 200        self,
 201        docker_compose_file: Path = Path("docker-compose.yml"),
 202        docker_compose_override_directory: Path = Path("docker-compose.override.d/"),
 203        docker_compose_task_file: Path = Path("docker-compose.tasks.yml"),
 204        docker_compose_task_override_directory: Path = Path(
 205            "docker-compose.tasks.override.d/"
 206        ),
 207    ):
 208        """
 209        Creates a new instance of an orchestrator for docker-compose-based applications.
 210
 211        Args:
 212            docker_compose_file (Path): Path to a docker compose file containing long-running services. Path is relative
 213                to the generated configuration directory.
 214            docker_compose_override_directory (Path, optional): Path to a directory containing any additional
 215                docker-compose override files. Overrides are applied in alphanumeric order of filename. Path is relative
 216                to the generated configuration directory.
 217            docker_compose_task_file (Path): Path to a docker compose file containing services to be run as short-lived
 218                tasks. Path is relative to the generated configuration directory.
 219            docker_compose_task_override_directory (Path): Path to a directory containing any additional
 220                docker-compose override files for services used as tasks. Path is relative to the generated
 221                configuration directory.
 222        """
 223        self.docker_compose_file = docker_compose_file
 224        self.docker_compose_override_directory = docker_compose_override_directory
 225        self.docker_compose_task_file = docker_compose_task_file
 226        self.docker_compose_task_override_directory = (
 227            docker_compose_task_override_directory
 228        )
 229
 230    def start(
 231        self, cli_context: CliContext, service_names: tuple[str, ...] = None
 232    ) -> CompletedProcess:
 233        command = ("up", "-d")
 234        if service_names is not None and len(service_names) > 0:
 235            command += service_names
 236        return self.__compose_service(cli_context, command)
 237
 238    def shutdown(
 239        self, cli_context: CliContext, service_names: tuple[str, ...] = None
 240    ) -> CompletedProcess:
 241        if service_names is not None and len(service_names) > 0:
 242            # We cannot use the 'down' command as it removes more than just the specified service (by design).
 243            # https://github.com/docker/compose/issues/5420
 244            # `-fsv` flags mean forcibly stop the container before removing, and delete attached anonymous volumes
 245            command = ("rm", "-fsv") + service_names
 246            return self.__compose_service(cli_context, command)
 247        return self.__compose_service(cli_context, ("down",))
 248
 249    def status(
 250        self, cli_context: CliContext, service_names: tuple[str, ...] = None
 251    ) -> CompletedProcess:
 252        command = ("ps", "-a")
 253        if service_names is not None and len(service_names) > 0:
 254            command += service_names
 255        return self.__compose_service(cli_context, command)
 256
 257    def task(
 258        self,
 259        cli_context: CliContext,
 260        service_name: str,
 261        extra_args: Iterable[str],
 262        detached: bool = False,
 263    ) -> CompletedProcess:
 264        command = ["run"]  # Command is: run [OPTIONS] --rm TASK [ARGS]
 265        if detached:
 266            command.append("-d")
 267        command.append("--rm")
 268        command.append(service_name)
 269        command.extend(list(extra_args))
 270        return self.__compose_task(cli_context, command)
 271
 272    def exec(
 273        self,
 274        cli_context: CliContext,
 275        service_name: str,
 276        command: Iterable[str],
 277        stdin_input: str = None,
 278        capture_output: bool = False,
 279    ) -> CompletedProcess:
 280        cmd = ["exec"]  # Command is: exec SERVICE COMMAND
 281        # If there's stdin_input being piped to the command, we need to provide
 282        # the -T flag to `docker-compose`: https://github.com/docker/compose/issues/7306
 283        if stdin_input is not None:
 284            cmd.append("-T")
 285        cmd.append(service_name)
 286        cmd.extend(list(command))
 287        return self.__compose_service(cli_context, cmd, stdin_input, capture_output)
 288
 289    def verify_service_names(
 290        self, cli_context: CliContext, service_names: tuple[str, ...]
 291    ) -> bool:
 292        if service_names is None or len(service_names) == 0:
 293            return True
 294        command = ["config", "--services"]
 295        result = self.__compose_service(cli_context, command, capture_output=True)
 296        if result.returncode != 0:
 297            error_msg = result.stderr.decode()
 298            logger.error(
 299                f"An unexpected error occured while verifying services. Error: {error_msg}"
 300            )
 301            return False
 302
 303        # Converts the byte type into list of names, and removes trailing empty string
 304        valid_service_names = result.stdout.decode().split("\n")[:-1]
 305        logger.debug("Valid Services: %s", ", ".join(valid_service_names))
 306        return service_name_verifier(service_names, valid_service_names)
 307

 311            context_settings=dict(ignore_unknown_options=True),
 312        )
 313        @click.pass_context
 314        @click.option(
 315            "--lines",
 316            "-n",
 317            help="Output the last NUM lines instead of all.",
 318            type=click.STRING,
 319            required=False,
 320            default="all",
 321        )
 322        @click.argument("service", nargs=-1, type=click.UNPROCESSED)
 323        def logs(ctx, lines, service):
 324            cli_context: CliContext = ctx.obj
 325            cli_context.get_configuration_dir_state().verify_command_allowed(
 326                AppcliCommand.SERVICE_LOGS
 327            )
 328            subcommand = ["logs", "--follow", f"--tail={lines}"]
 329            subcommand.extend(service)
 330            result = self.__compose_service(cli_context, subcommand)
 331            sys.exit(result.returncode)
 332
 333        return logs

 430        self, cli_context: CliContext, service_names: tuple[str, ...] = None
 431    ) -> CompletedProcess:
 432        if service_names is not None and len(service_names) > 0:
 433            logger.error(
 434                "Docker Swarm orchestrator cannot start individual services. Attempted to start [%s].",
 435                service_names,
 436            )
 437            return CompletedProcess(args=None, returncode=1)
 438
 439        subcommand = ["deploy"]
 440        compose_files = decrypt_docker_compose_files(
 441            cli_context,
 442            self.docker_compose_file,
 443            self.docker_compose_override_directory,
 444        )
 445        if len(compose_files) == 0:
 446            logger.error(
 447                "No valid docker compose files were found. Expected file [%s] or files in directory [%s]",
 448                self.docker_compose_file,
 449                self.docker_compose_override_directory,
 450            )
 451            return CompletedProcess(args=None, returncode=1)
 452        for compose_file in compose_files:
 453            subcommand.extend(("--compose-file", str(compose_file)))
 454
 455        return self.__docker_stack(cli_context, subcommand)

 835    def __generate_values_args(self, generated_configuration_dir: Path) -> list[str]:
 836        """Recursively takes all the values files in the directories and generates an args array to
 837        pass to helm through either `--values` or `--set-file` (depending on the location).
 838        e.g:
 839
 840            ["--values", "values.yaml", "--set-file", "path.to.foo=path/to/foo.yaml"...
 841
 842        Args:
 843            generated_configuration_dir (Path): Generated configuration directory form the cli object, e.g:
 844                `cli_context.get_generated_configuration_dir()`
 845
 846        Returns:
 847            list[str]: The arg list.
 848        """
 849        arg_list = []
 850
 851        # Create all `--values` args.
 852        values_dir = generated_configuration_dir / self.helm_set_values_dir
 853        values = [
 854            file
 855            for file in list(values_dir.rglob("*"))
 856            if file.suffix in [".yml", ".yaml"]
 857        ]
 858        for file in values:
 859            arg_list.append("--values")
 860            arg_list.append(str(file))
 861
 862        # Create all `--set-file` args.
 863        values_files_dir = generated_configuration_dir / self.helm_set_files_dir
 864        values_files = [
 865            file for file in list(values_files_dir.rglob("*")) if file.is_file()
 866        ]
 867        for file in values_files:
 868            # NOTE: Make path relative to `helm_values_files_dir` so we know which helm key to set it as.
 869            relative_file = file.relative_to(
 870                generated_configuration_dir / self.helm_set_files_dir
 871            )
 872            # Get the helm key in `dot.notation.to.value`
 873            key = ".".join(
 874                [*relative_file.parent.parts, relative_file.stem.split(".")[0]]
 875            )
 876            arg_list.append("--set-file")
 877            arg_list.append(f"{key}={file}")
 878
 879        return arg_list

 982    docker_compose_override_directory_relative_path: Path,
 983) -> List[Path]:
 984    """Decrypt docker-compose and docker-compose override files.
 985
 986    Args:
 987        cli_context (CliContext): The current CLI context.
 988        docker_compose_file_relative_path (Path): The relative path to the docker-compose file. Path is relative to the
 989            generated configuration directory.

 991            docker-compose override files. Path is relative to the generated configuration directory.
 992
 993    Returns:
 994        List[Path]: sorted list of absolute paths to decrypted docker-compose files. The first path is the decrypted
 995            docker-compose file, and the rest of the paths are the alphanumerically sorted docker compose override
 996            files in the docker compose override directory.
 997    """
 998
 999    compose_files = []
1000
1001    if docker_compose_file_relative_path is not None:
1002        docker_compose_file = cli_context.get_generated_configuration_dir().joinpath(
1003            docker_compose_file_relative_path
1004        )
1005        if os.path.isfile(docker_compose_file):
1006            compose_files.append(docker_compose_file)
1007
1008    if docker_compose_override_directory_relative_path is not None:
1009        docker_compose_override_directory = (
1010            cli_context.get_generated_configuration_dir().joinpath(
1011                docker_compose_override_directory_relative_path
1012            )
1013        )
1014        if os.path.isdir(docker_compose_override_directory):
1015            docker_compose_override_files: List[Path] = [
1016                Path(os.path.join(docker_compose_override_directory, file))
1017                for file in os.listdir(docker_compose_override_directory)
1018                if os.path.isfile(os.path.join(docker_compose_override_directory, file))
1019            ]
1020
1021            if len(docker_compose_override_files) > 0:
1022                docker_compose_override_files.sort()
1023                logger.debug(
1024                    "Detected docker compose override files [%s]",
1025                    docker_compose_override_files,
1026                )
1027                compose_files.extend(docker_compose_override_files)
1028
1029    # decrypt files if key is available
1030    key_file = cli_context.get_key_file()
1031    decrypted_files = [
1032        decrypt_file(encrypted_file, key_file) for encrypted_file in compose_files
1033    ]
1034    return decrypted_files
