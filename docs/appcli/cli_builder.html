 65def create_cli(configuration: Configuration, desired_environment: Dict[str, str] = {}):
 66    """Build the CLI to be run
 67
 68    Args:
 69        configuration (Configuration): the application's configuration settings
 70    """
 71
 72    # We currently only support the `NullOrchestrator` on Windows.
 73    # TODO APED-69: Add support for Docker orchestrators on Windows.
 74    if IS_PLATFORM_WINDOWS and not isinstance(
 75        configuration.orchestrator, NullOrchestrator
 76    ):
 77        error_msg = f"Unsupported Windows orchestrator `{type(configuration.orchestrator).__name__}`. Only `NullOrchestrator` is supported on Windows systems."
 78        error_and_exit(error_msg)
 79
 80    APP_NAME = configuration.app_name
 81    APP_NAME_SLUG = configuration.app_name_slug
 82    APP_NAME_SLUG_UPPER = APP_NAME_SLUG.upper()
 83    APP_VERSION = os.environ.get("APP_VERSION", "latest")
 84    IS_DEV_MODE: bool = f"{APP_NAME_SLUG_UPPER}_DEV_MODE" in os.environ
 85
 86    # Details of the user who ran the CLI app.
 87    # NOTE: We use `getpass` to get the username because it works on both Linux and Windows.
 88    CLI_USER = os.environ.get(f"{APP_NAME_SLUG_UPPER}_CLI_USER", getpass.getuser())
 89    # Windows does not support uid/gid so default to 1000 if on Windows.
 90    system_uid = "1000" if IS_PLATFORM_WINDOWS else str(os.getuid())
 91    system_gid = "1000" if IS_PLATFORM_WINDOWS else str(os.getgid())
 92    CLI_UID = os.environ.get(f"{APP_NAME_SLUG_UPPER}_CLI_UID", system_uid)
 93    CLI_GID = os.environ.get(f"{APP_NAME_SLUG_UPPER}_CLI_GID", system_gid)
 94
 95    # Mandatory environment variables this script will set.
 96    ENV_VAR_CONFIG_DIR = f"{APP_NAME_SLUG_UPPER}_CONFIG_DIR"
 97    ENV_VAR_GENERATED_CONFIG_DIR = f"{APP_NAME_SLUG_UPPER}_GENERATED_CONFIG_DIR"
 98    ENV_VAR_DATA_DIR = f"{APP_NAME_SLUG_UPPER}_DATA_DIR"
 99    ENV_VAR_BACKUP_DIR = f"{APP_NAME_SLUG_UPPER}_BACKUP_DIR"
100    ENV_VAR_ENVIRONMENT = f"{APP_NAME_SLUG_UPPER}_ENVIRONMENT"
101
102    DEV_MODE_VARIABLES = {}
103    if IS_DEV_MODE:
104        with wrap_dev_mode():
105            # It is recommended to introduce some randomness to temporary file paths.
106            # This prevents attackers knowing the directory, where they can perfrom attacks such as:
107            # Per-creating synlinks there, DoS and file overwrites, etc.
108            # See https://cwe.mitre.org/data/definitions/377.html
109            # NOTE: We are okay to take the risk here as this is local-dev only.
110            install_dir_base = Path("/tmp/") / APP_NAME_SLUG.lower()  # nosec B108
111            environment = "local-dev"
112            install_dir = install_dir_base / environment
113            install_dir.mkdir(parents=True, exist_ok=True)
114
115            overrides = {
116                f"{APP_NAME_SLUG_UPPER}_CLI_DEBUG": True,
117                f"{APP_NAME_SLUG_UPPER}_CLI_ENVIRONMENT": environment,
118                # If running `install` command, override install directory.
119                f"{APP_NAME_SLUG_UPPER}_CLI_INSTALL_INSTALL_DIR": install_dir_base,
120                # If running any other command, override relevant directories.
121                f"{APP_NAME_SLUG_UPPER}_CLI_CONFIGURATION_DIR": install_dir / "conf",
122                f"{APP_NAME_SLUG_UPPER}_CLI_DATA_DIR": install_dir / "data",
123                f"{APP_NAME_SLUG_UPPER}_CLI_BACKUP_DIR": install_dir / "backup",
124            }
125
126            logger.info("Overriding CLI options via environment variables")
127            for key, value in overrides.items():
128                logger.info(f"  {key}={value}")
129                os.environ[key] = str(value)
130
131            # Any environment variable beginning with `{APP_NAME}_DEV_MODE_*`
132            # is loaded up as a `DEV_MODE` variable.
133            logger.info("Detecting and setting `DEV_MODE` variables.")
134            for key, value in os.environ.items():
135                if key.startswith(f"{APP_NAME_SLUG_UPPER}_DEV_MODE_"):
136                    logger.info(
137                        f"Found the `[{key}]` dev variable. Setting to `{value}`."
138                    )
139                    DEV_MODE_VARIABLES[key] = value

142    # CREATE_CLI: LOGIC
143    # --------------------------------------------------------------------------
144
145    default_commands = {}
146    for cli_class in (
147        ConfigureCli,
148        DebugCli,
149        EncryptCli,
150        InitCli,
151        InstallCli,
152        LauncherCli,
153        MigrateCli,
154        ServiceCli,
155        TaskCli,
156        BackupManagerCli,
157        VersionCli,
158    ):
159        commands = cli_class(configuration).commands
160        default_commands.update(**commands)
161
162    # Remove any default actions which the orchestrator does not support.
163    for disabled_command in configuration.orchestrator.get_disabled_commands():
164        default_commands.pop(disabled_command)

300            ["Data directory", f"{cli_context.data_dir}"],

302            ["Current user", f"{CLI_USER} (uid: {CLI_UID} / gid: {CLI_GID})"],
303        ]
304
305        # Print out the configuration values as an aligned table
306        details = tabulate(table, colalign=("right",))
307        logger.info(f"Current context:\n{details}\n")
308        if additional_data_dir:
309            logger.info(
310                "Additional data directories:\n\n%s\n",
311                tabulate(
312                    additional_data_dir,
313                    headers=["Environment Variable", "Path"],
314                    colalign=("right",),
315                ),
316            )
317        if additional_env_var:
318            logger.info(
319                "Additional environment variables:\n\n%s\n",
320                tabulate(
321                    additional_env_var,
322                    headers=["Environment Variable", "Value"],
323                    colalign=("right",),
324                ),
325            )
326
327        if IS_DEV_MODE:
328            with wrap_dev_mode():
329                # Auto configure init/apply if not specified.
330                if ctx.invoked_subcommand not in ("configure",):
331                    configure_cli = cli_context.commands["configure"]
332                    try:

334                        ctx.invoke(configure_cli.commands["init"])
335                    except SystemExit:
336                        # At completion, the invoked command tries to exit the script, so we have to catch

338                        pass

341                        logger.info("Auto running `configure apply`")

343                    except SystemExit:
344                        # At completion, the invoked command tries to exit the script, so we have to catch
345                        # the SystemExit.
346                        pass

349        """Run the entry-point click CLI command"""

355    def __check_docker_socket():
356        """Check that the docker socket exists, and exit if it does not"""
357        if not os.path.exists("/var/run/docker.sock"):
358            error_msg = """Docker socket not present. Please launch with a mounted /var/run/docker.sock"""
359            error_and_exit(error_msg)

361    def __set_environment(
362        cli_context: CliContext, desired_environment: Dict[str, str] = {}
363    ):
364        mandatory_environment = {
365            ENV_VAR_CONFIG_DIR: cli_context.configuration_dir,
366            ENV_VAR_GENERATED_CONFIG_DIR: cli_context.get_generated_configuration_dir(),
367            ENV_VAR_DATA_DIR: cli_context.data_dir,
368            ENV_VAR_BACKUP_DIR: cli_context.backup_dir,
369            ENV_VAR_ENVIRONMENT: cli_context.environment,
370        }
371
372        final_environment = mandatory_environment | desired_environment
373
374        # Attempt to set desired environment.
375        initialised_environment = {}
376        for k, v in final_environment.items():
377            if v is None:
378                logger.warning(f"Environment variable `{k}` has not been set")
379            else:
380                logger.debug(f"Exporting environment variable: {k}={v}")
381                os.environ[k] = str(v)
382                initialised_environment[k] = str(v)
383        if len(initialised_environment) != len(final_environment):
384            error_and_exit(
385                "Could not set desired environment. Please ensure specified environment variables are set."
386            )
387
388    def __check_environment():
389        """Confirm that mandatory environment variables and additional data directories are defined."""
390
391        launcher_set_mandatory_env_vars = [
392            ENV_VAR_CONFIG_DIR,
393            ENV_VAR_GENERATED_CONFIG_DIR,
394            ENV_VAR_DATA_DIR,
395            ENV_VAR_BACKUP_DIR,
396            ENV_VAR_ENVIRONMENT,
397        ]
398
399        launcher_env_vars_set = check_environment_variable_defined(
400            launcher_set_mandatory_env_vars,
401            "Mandatory environment variable [%s] not defined. This should be set within the script generated with the 'launcher' command.",
402            "Cannot run without all mandatory environment variables defined",
403        )
404
405        additional_env_vars_set = check_environment_variable_defined(
406            configuration.mandatory_additional_env_variables,
407            'Mandatory additional environment variable [%s] not defined. When running the \'launcher\' command, define with:\n\t--additional-env-var "%s"="<value>"',
408            "Cannot run without all mandatory additional environment variables defined",
409        )
410
411        additional_data_dir_env_vars_set = check_environment_variable_defined(
412            configuration.mandatory_additional_data_dirs,
413            'Mandatory additional data directory [%s] not defined. When running the \'launcher\' command, define with:\n\t--additional-data-dir "%s"="</path/to/dir>"',
414            "Cannot run without all mandatory additional data directories defined",
415        )
416
417        if not (
418            launcher_env_vars_set
419            and additional_env_vars_set
420            and additional_data_dir_env_vars_set
421        ):
422            error_and_exit(
423                "Some mandatory environment variables weren't set. See error messages above."
424            )
425        logger.info("All required environment variables are set.")

427    def __copy_config_to_home_dir(config_src_dir: Path) -> None:
428        """Copies the .generated `cli/home` directory into `/root` in the container.
429
430        The program might generate data in the `/root` directory which we do not want to lose on restart.
431        We therefore have to merge user provided files from `cli/home` into the container.
432
433        Args:
434            config_src_dir (Path): The location of the user provided home dir.
435        """
436        # Do not copy the users data if we are in dev mode as we are probably not in a container.
437        if IS_DEV_MODE:
438            return
439
440        if config_src_dir.exists() and config_src_dir.is_dir():
441            logger.info(
442                f"Copying `{config_src_dir}` to `{CONTAINER_HOME_DIR}` container directory."
443            )
444            shutil.copytree(config_src_dir, CONTAINER_HOME_DIR, dirs_exist_ok=True)
445
446    def check_environment_variable_defined(
447        env_variables: Iterable[str], error_message_template: str, exit_message: str
448    ) -> bool:
449        """Check if environment variables are defined
450
451        Args:
452            env_variables (Iterable[str]): the environment variables to check
453            error_message_template (str): a template for the error message
454            exit_message (str): the exit message on error
455
456        Returns:
457            [bool]: True if all environment variables are defined, otherwise False.
458        """
459        result = True
460        for env_variable in env_variables:
461            value = os.environ.get(env_variable)
462            if value is None:
463                logger.error(error_message_template, env_variable, env_variable)
464                result = False
465            else:
466                logger.debug(
467                    f"Confirmed environment variable is set: {env_variable}={value}"
468                )
469        if not result:
470            logger.error(exit_message)
471
472        return result
473
474    for command in default_commands.values():
475        cli.add_command(command)
476
477    for command in configuration.custom_commands:
478        cli.add_command(command)
479
480    return run

491# allow required options to be ignored on certain subcommands
492# see: https://stackoverflow.com/a/51235564/3602961
493class NotRequiredOn(click.Option):
494    def __init__(self, *args, **kwargs):
495        self.not_required_on = kwargs.pop("not_required_on")
496        if not self.not_required_on:
497            raise AttributeError("'not_required_on' parameter required")
498        kwargs["help"] = (
499            kwargs.get("help", "")
500            + f"  [required] unless subcommand is one of: {self.not_required_on}."
501        )
502        super(NotRequiredOn, self).__init__(*args, **kwargs)
503
504    def handle_parse_result(self, ctx, opts, args):
505        subcommand = args[0] if len(args) > 0 else ""
506        if subcommand in self.not_required_on:
507            self.prompt = None
508            return super(NotRequiredOn, self).handle_parse_result(ctx, opts, args)
509
510        # Delegate to super to try and populate value from opts/env vars, etc.
511        result = super(NotRequiredOn, self).handle_parse_result(ctx, opts, args)
512        # First item in the tuple is the parsed value for the option. Fail if missing.
513        if result[0] is None:
514            options = " / ".join([f"'{x}'" for x in self.opts])
515            raise click.UsageError(f"Error: Missing option {options}.")
516
517        return result

454            error_message_template (str): a template for the error message

Options are usually optional values on the command line and have some extra features that arguments don't have.

All other parameters are passed onwards to the parameter constructor.

Parameters
  • show_default: Show the default value for this option in its help text. Values are not shown by default, unless Context.show_default is True. If this value is a string, it shows that string in parentheses instead of the actual value. This is particularly useful for dynamic options. For single option boolean flags, the default remains hidden if its value is False.
  • show_envvar: Controls if an environment variable should be shown on the help page. Normally, environment variables are not shown.
  • prompt: If set to True or a non empty string then the user will be prompted for input. If set to True the prompt will be the option name capitalized.
  • confirmation_prompt: Prompt a second time to confirm the value if it was prompted for. Can be set to a string instead of True to customize the message.
  • prompt_required: If set to False, the user will be prompted for input only when the option was specified as a flag without a value.
  • hide_input: If this is True then the input on the prompt will be hidden from the user. This is useful for password input.
  • is_flag: forces this option to act as a flag. The default is auto detection.
  • flag_value: which value should be used for this flag if it's enabled. This is set to a boolean automatically if the option string contains a slash to mark two options.
  • multiple: if this is set to True then the argument is accepted multiple times and recorded. This is similar to nargs in how it works but supports arbitrary number of arguments.
  • count: this flag makes an option increment an integer.
  • allow_from_autoenv: if this is enabled then the value of this parameter will be pulled from an environment variable in case a prefix is defined on the context.
  • help: the help string.
  • hidden: hide this option from help outputs.
